import random

def fitness(x):
    return x**2

num_nests = 10
num_generations = 20
pa = 0.25
lb, ub = -10, 10

nests = [random.uniform(lb, ub) for _ in range(num_nests)]
best_nest = min(nests, key=fitness)

for gen in range(1, num_generations + 1):
    for i in range(num_nests):
        step_size = random.uniform(-1, 1)
        new_nest = nests[i] + step_size
        if new_nest < lb: new_nest = lb
        if new_nest > ub: new_nest = ub
        if fitness(new_nest) < fitness(nests[i]):
            nests[i] = new_nest
    for i in range(num_nests):
        if random.random() < pa:
            nests[i] = random.uniform(lb, ub)
    current_best = min(nests, key=fitness)
    if fitness(current_best) < fitness(best_nest):
        best_nest = current_best
    print(f"Generation {gen}: Best x = {round(best_nest,4)}, f(x) = {round(fitness(best_nest),4)}")

print("\nFinal Best Solution:")
print(f"x = {round(best_nest,4)}, f(x) = {round(fitness(best_nest),4)}")
