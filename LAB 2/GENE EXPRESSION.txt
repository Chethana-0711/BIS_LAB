import random


def fitness_function(x):
    return x ** 2


POP_SIZE = 6           
GENERATIONS = 5        
MUTATION_RATE = 0.2    
X_RANGE = (-10, 10)    


population = [random.uniform(*X_RANGE) for _ in range(POP_SIZE)]

print("Initial Population:")
print(population)


for gen in range(1, GENERATIONS + 1):
   
    fitness = [fitness_function(x) for x in population]
    
    
    sorted_pop = [x for _, x in sorted(zip(fitness, population))]
    parents = sorted_pop[:2]  
    
    
    offspring = []
    while len(offspring) < POP_SIZE - len(parents):
        p1, p2 = random.sample(parents, 2)
        alpha = random.random()
        child = alpha * p1 + (1 - alpha) * p2
        offspring.append(child)
    
    
    for i in range(len(offspring)):
        if random.random() < MUTATION_RATE:
            offspring[i] += random.uniform(-1, 1)  

    
    population = parents + offspring

    
    best_x = min(population, key=lambda x: fitness_function(x))
    best_f = fitness_function(best_x)
    print(f"\nGeneration {gen}:")
    print(f"Population: {[round(x, 3) for x in population]}")
    print(f"Best x = {round(best_x, 4)}, f(x) = {round(best_f, 4)}")

print("\nâœ… Final Best Solution:")
print(f"x = {round(best_x, 4)}, f(x) = {round(best_f, 4)}")
